    function web(type, id)
    {
        const out = new WebSocket("");
        return new Promise(then => out.onopen = () => {
            send(out, { type, id });
            then(out);
        });
    }

    function send(socket, obj)
    {
        return socket.send(JSON.stringify(obj));
    }

    async function main()
    {
        a = await web("admin", "d7cdeba7-5d07-11eb-9d07-16fe402bc5a4");
